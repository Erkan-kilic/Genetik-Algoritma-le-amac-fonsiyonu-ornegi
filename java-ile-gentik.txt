    public static   int      OnlukTabanaCevir(String BinaryDizi){

        int i=0,sonuc = 0,x,k[]=new int[100];

        x =Integer.parseInt(BinaryDizi);

        while(x>=1){
            i++;
            k[i]=(int) (((x%10) * Math.pow(2,i))/2);
            x=x/10;
            sonuc= sonuc + k[i];

        }


        return sonuc;

    }
